		\begin{tikzpicture}
			\node [mybox] (box){%
				\begin{minipage}{0.46\textwidth}
					
					$X \sim Bernoulli(p)$ \\
					$P(X=k) = p^k(1-p)^{1-k}$ \\
					
					\begin{tabular}{lp{8cm} l}
						Mean: &$\mu = E(X) = p$ \\
						Variance: &$\sigma^2 = Var(X) = p(1-p)$\\
					\end{tabular}\\
				\end{minipage}
			};
			%------------  Bernoulli Distribution Header ---------------------

				\begin{minipage}{0.46\textwidth}
				
					$X \sim Bin(n, p)$\\
					$P(X=k) = {n \choose k} p^k(1-p)^{n-k}$\\
				
					\begin{tabular}{lp{8cm} l}
						Mean: &$\mu = E(X) = np$ \\
						Variance: &$\sigma^2 = Var(X) = np(1-p)$\\
					\end{tabular}
				\end{minipage}
			};
			%------------  Binomial Distribution Header ---------------------
			\node[fancytitle, right=10pt] at (box.north west) {Binomial Distribution};

					$P(X=k)=\frac{\lambda^k e^{-\lambda}}{k!}$\\
					
					\begin{tabular}{lp{8cm} l}
						Mean: &$\mu = E(X) = \lambda$ \\
						Variance: &$\sigma^2 = Var(X) = \lambda$\\

					\begin{tabular}{lp{8cm} l}

					$P(X = k) = {k-1 \choose r-1}(1-p)^{k-r}p^r$ \\
					
					\begin{tabular}{lp{8cm} l}
						Mean: &$\mu = E(X) = \frac{r}{p}$ \\
						Variance: &$\sigma^2 = Var(X) = \frac{r(1-p)}{p^2}$\\

					$P(X = k) = \frac{{r \choose k}{n-r \choose m-k}}{{n \choose m}}$ \\
					
					\begin{tabular}{lp{8cm} l}

						Variance: &$\sigma^2 = Var(X) = \frac{n-m}{n-1} \cdot m \cdot \frac{r}{n} \cdot \frac{n-r}{n}$\\
					\end{tabular}\\
				\end{minipage}
